# Tax Page + Simulator + F24 Generation — Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Build a tax overview page (`/taxes`), a tax simulator (`/taxes/simulator`), and F24 PDF generation with payment tracking for forfettario freelancers.

**Architecture:** The tax overview uses a new `GET /api/taxes/overview` endpoint that aggregates invoices + profile + payment records (from existing `tax_periods` table). F24 PDFs are generated server-side with `pdf-lib` filling the editable `F24_editabile.pdf` template. The simulator runs calculations client-side using the shared tax engine. Payment tracking uses a new `tax_payments` table (simpler than the existing `tax_periods` schema).

**Tech Stack:** Fastify API, pdf-lib, Drizzle ORM, React + shadcn/ui, TanStack Query, `@fatturino/shared` tax engine

```typescript
// Add to imports
import { taxPayments } from "../db/schema.js";

// Add this route inside the existing plugin function
app.get<{ Querystring: { anno: string } }>(
  "/api/taxes/overview",
  async (request, reply) => {
    const userId = getUserId(request);
    const anno = parseInt(request.query.anno, 10);
    if (!anno || anno < 1900 || anno > 2100) {
      return reply.status(400).send({ error: "Invalid anno parameter" });
    }

    // Fetch invoices for the year
    const yearInvoices = await db
      .select()
      .from(invoices)
      .where(and(eq(invoices.userId, userId), eq(invoices.anno, anno)));

    // Fetch profile
    const profiles = await db
      .select()
      .from(userProfiles)
      .where(eq(userProfiles.userId, userId));
    const profile = profiles[0] ?? null;

    // Fetch payment records
    const payments = await db
      .select()
      .from(taxPayments)
      .where(and(eq(taxPayments.userId, userId), eq(taxPayments.anno, anno)));

    // Aggregate
    const nonDraft = yearInvoices.filter((i) => i.stato !== "bozza");
    const totalRevenue = nonDraft.reduce(
      (sum, i) => sum + parseFloat(String(i.totaleDocumento)),
      0
    );

    const profileIncomplete = !profile?.codiceAteco || !profile?.annoInizioAttivita;

    let tax = null;
    let inps = null;
    let f24 = null;

    if (!profileIncomplete && profile) {
      try {
        const inpsResult = calcolaInps({
          fatturato: totalRevenue,
          codiceAteco: profile.codiceAteco,
          gestione: profile.gestioneInps as GestioneInps,
        });
        const taxResult = calcolaImposta({
          fatturato: totalRevenue,
          codiceAteco: profile.codiceAteco,
          contributiInpsVersati: inpsResult.totaleDovuto,
          annoInizioAttivita: profile.annoInizioAttivita,
          annoFiscale: anno,
        });

        // Use actual accontiVersati from payments
        const accontiPaid = payments
          .filter((p) => p.deadline !== "saldo" && p.amountPaid)
          .reduce((sum, p) => sum + parseFloat(String(p.amountPaid)), 0);

        const f24Result = calcolaAccontoSaldo({
          impostaDovuta: taxResult.impostaDovuta,
          accontiVersati: accontiPaid,
          anno,
        });

        tax = {
          coefficienteRedditivita: taxResult.coefficienteRedditivita,
          redditoLordo: taxResult.redditoLordo,
          redditoImponibile: taxResult.redditoImponibile,
          aliquota: taxResult.aliquota,
          isStartup: taxResult.isStartup,
          impostaDovuta: taxResult.impostaDovuta,
        };
        inps = {
          gestione: profile.gestioneInps,
          baseImponibile: inpsResult.baseImponibile,
          contributoFisso: inpsResult.contributoFisso,
          contributoEccedenza: inpsResult.contributoEccedenza,
          totaleDovuto: inpsResult.totaleDovuto,
        };
        f24 = f24Result;
      } catch {
        // Invalid ATECO or other error — leave as null
      }
    }

    // Build payment status array
    const deadlines = ["primo_acconto", "secondo_acconto", "saldo"] as const;
    const now = new Date();
    const deadlineDates: Record<string, Date> = {
      primo_acconto: new Date(anno, 5, 30),  // June 30
      secondo_acconto: new Date(anno, 10, 30), // Nov 30
      saldo: new Date(anno + 1, 5, 30),       // June 30 next year
    };

    const paymentStatus = deadlines.map((d) => {
      const record = payments.find((p) => p.deadline === d);
      const dueAmount = d === "primo_acconto"
        ? f24?.primoAcconto ?? 0
        : d === "secondo_acconto"
          ? f24?.secondoAcconto ?? 0
          : f24?.saldo ?? 0;
      const isPaid = record?.amountPaid != null;
      const isOverdue = !isPaid && now > deadlineDates[d];

      return {
        deadline: d,
        amountDue: dueAmount,
        amountPaid: record?.amountPaid ? parseFloat(String(record.amountPaid)) : null,
        datePaid: record?.datePaid ?? null,
        status: isPaid ? "paid" : isOverdue ? "overdue" : "pending",
      };
    });

    return {
      anno,
      totalRevenue,
      tax,
      inps,
      f24,
      payments: paymentStatus,
      profileIncomplete,
    };
  }
);
```

```typescript
const paymentSchema = z.object({
  anno: z.number().int().min(1900).max(2100),
  deadline: z.enum(["primo_acconto", "secondo_acconto", "saldo"]),
  amountPaid: z.number().nonnegative(),
  datePaid: z.string().regex(/^\d{4}-\d{2}-\d{2}$/),
});

app.post("/api/taxes/payments", async (request, reply) => {
  const userId = getUserId(request);
  const parsed = paymentSchema.safeParse(request.body);
  if (!parsed.success) {
    return reply.status(400).send({ error: "Invalid payment data", details: parsed.error.issues });
  }

  const { anno, deadline, amountPaid, datePaid } = parsed.data;

  // Upsert — update if exists, insert if not
  const existing = await db
    .select()
    .from(taxPayments)
    .where(
      and(
        eq(taxPayments.userId, userId),
        eq(taxPayments.anno, anno),
        eq(taxPayments.deadline, deadline)
      )
    );

  if (existing.length > 0) {
    const [updated] = await db
      .update(taxPayments)
      .set({
        amountPaid: String(amountPaid),
        datePaid,
        updatedAt: new Date(),
      })
      .where(eq(taxPayments.id, existing[0].id))
      .returning();
    return updated;
  }

  // Compute amountDue from tax engine (need invoices + profile)
  const yearInvoices = await db
    .select()
    .from(invoices)
    .where(and(eq(invoices.userId, userId), eq(invoices.anno, anno)));
  const profiles = await db
    .select()
    .from(userProfiles)
    .where(eq(userProfiles.userId, userId));
  const profile = profiles[0];

  let amountDue = 0;
  if (profile?.codiceAteco && profile?.annoInizioAttivita) {
    try {
      const nonDraft = yearInvoices.filter((i) => i.stato !== "bozza");
      const totalRevenue = nonDraft.reduce(
        (sum, i) => sum + parseFloat(String(i.totaleDocumento)), 0
      );
      const inpsResult = calcolaInps({
        fatturato: totalRevenue,
        codiceAteco: profile.codiceAteco,
        gestione: profile.gestioneInps as GestioneInps,
      });
      const taxResult = calcolaImposta({
        fatturato: totalRevenue,
        codiceAteco: profile.codiceAteco,
        contributiInpsVersati: inpsResult.totaleDovuto,
        annoInizioAttivita: profile.annoInizioAttivita,
        annoFiscale: anno,
      });
      const f24 = calcolaAccontoSaldo({ impostaDovuta: taxResult.impostaDovuta, accontiVersati: 0, anno });
      amountDue = deadline === "primo_acconto" ? f24.primoAcconto
        : deadline === "secondo_acconto" ? f24.secondoAcconto
        : f24.saldo;
    } catch {
      // leave as 0
    }
  }

  const [inserted] = await db
    .insert(taxPayments)
    .values({
      userId,
      anno,
      deadline,
      amountDue: String(amountDue),
      amountPaid: String(amountPaid),
      datePaid,
    })
    .returning();

  return inserted;
});
```

```typescript
import { PDFDocument } from "pdf-lib";
import { readFile } from "node:fs/promises";
import { join, dirname } from "node:path";
import { fileURLToPath } from "node:url";
import { CODICI_TRIBUTO_IMPOSTA } from "@fatturino/shared";

const __dirname = dirname(fileURLToPath(import.meta.url));

// Resolve template path — goes from apps/api/src/services/ up to repo root docs/
const TEMPLATE_PATH = join(__dirname, "../../../../docs/F24_editabile.pdf");

export type F24Deadline = "primo_acconto" | "secondo_acconto" | "saldo";

interface F24FillParams {
  codiceFiscale: string;
  ragioneSociale: string;
  anno: number;
  deadline: F24Deadline;
  amount: number;
}

const DEADLINE_CODICE: Record<F24Deadline, string> = {
  primo_acconto: CODICI_TRIBUTO_IMPOSTA.ACCONTO_PRIMO,   // "1790"
  secondo_acconto: CODICI_TRIBUTO_IMPOSTA.ACCONTO_SECONDO, // "1791"
  saldo: CODICI_TRIBUTO_IMPOSTA.SALDO,                   // "1792"
};

export async function generateF24Pdf(params: F24FillParams): Promise<Buffer> {
  const templateBytes = await readFile(TEMPLATE_PATH);
  const pdfDoc = await PDFDocument.load(templateBytes);
  const form = pdfDoc.getForm();

  // Fill codice fiscale (cf1–cf16, one char each)
  const cf = params.codiceFiscale.toUpperCase();
  for (let i = 0; i < Math.min(cf.length, 16); i++) {
    try {
      const field = form.getTextField(`cf${i + 1}`);
      field.setText(cf[i]);
    } catch {
      // Field might not exist
    }
  }

  // Fill ragione sociale
  try {
    form.getTextField("ragsociale").setText(params.ragioneSociale);
  } catch {
    // Field might not exist
  }

  // Fill Sezione Erario — row 1
  const codiceTributo = DEADLINE_CODICE[params.deadline];
  try {
    form.getTextField("codtrib1").setText(codiceTributo);
  } catch {}

  // Anno di riferimento
  try {
    form.getTextField("annorif1").setText(String(params.anno));
  } catch {}

  // Importo a debito versato (euros + cents)
  const amountStr = params.amount.toFixed(2);
  try {
    form.getTextField("impvers1").setText(amountStr);
  } catch {}

  // Totale A (same as single row amount)
  try {
    form.getTextField("tota").setText(amountStr);
  } catch {}

  // Saldo A-B
  try {
    form.getTextField("salab").setText(amountStr);
  } catch {}

  // Flatten to make fields non-editable in output
  form.flatten();

  const pdfBytes = await pdfDoc.save();
  return Buffer.from(pdfBytes);
}
```

```typescript
import { generateF24Pdf, type F24Deadline } from "../services/f24-pdf.js";

const VALID_DEADLINES = ["primo-acconto", "secondo-acconto", "saldo"] as const;

app.get<{ Params: { anno: string; deadline: string } }>(
  "/api/taxes/:anno/f24/:deadline",
  async (request, reply) => {
    const userId = getUserId(request);
    const anno = parseInt(request.params.anno, 10);
    const deadlineParam = request.params.deadline;

    if (!anno || !VALID_DEADLINES.includes(deadlineParam as any)) {
      return reply.status(400).send({ error: "Invalid anno or deadline" });
    }

    // Map URL param to DB enum
    const deadline = deadlineParam.replace("-", "_") as F24Deadline;

    // Get profile for codice fiscale and ragione sociale
    const profiles = await db
      .select()
      .from(userProfiles)
      .where(eq(userProfiles.userId, userId));
    const profile = profiles[0];

    if (!profile) {
      return reply.status(400).send({ error: "Profile required to generate F24" });
    }

    // Compute amount for this deadline
    const yearInvoices = await db
      .select()
      .from(invoices)
      .where(and(eq(invoices.userId, userId), eq(invoices.anno, anno)));

    const nonDraft = yearInvoices.filter((i) => i.stato !== "bozza");
    const totalRevenue = nonDraft.reduce(
      (sum, i) => sum + parseFloat(String(i.totaleDocumento)), 0
    );

    let amount = 0;
    try {
      const inpsResult = calcolaInps({
        fatturato: totalRevenue,
        codiceAteco: profile.codiceAteco,
        gestione: profile.gestioneInps as GestioneInps,
      });
      const taxResult = calcolaImposta({
        fatturato: totalRevenue,
        codiceAteco: profile.codiceAteco,
        contributiInpsVersati: inpsResult.totaleDovuto,
        annoInizioAttivita: profile.annoInizioAttivita,
        annoFiscale: anno,
      });
      const f24 = calcolaAccontoSaldo({ impostaDovuta: taxResult.impostaDovuta, accontiVersati: 0, anno });
      amount = deadline === "primo_acconto" ? f24.primoAcconto
        : deadline === "secondo_acconto" ? f24.secondoAcconto
        : f24.saldo;
    } catch {
      return reply.status(400).send({ error: "Unable to compute tax for this year" });
    }

    const pdfBuffer = await generateF24Pdf({
      codiceFiscale: profile.codiceFiscale,
      ragioneSociale: profile.ragioneSociale,
      anno,
      deadline,
      amount,
    });

    const deadlineLabel = deadlineParam.replace("-", "_");
    return reply
      .header("Content-Type", "application/pdf")
      .header("Content-Disposition", `attachment; filename="F24_${anno}_${deadlineLabel}.pdf"`)
      .send(pdfBuffer);
  }
);
```

```typescript
  profileIncomplete: boolean;
}

export function useTaxOverview(anno: number) {
  return useQuery<TaxOverview>({
    queryKey: ["taxes", anno],
    queryFn: () => api.get<TaxOverview>(`/taxes/overview?anno=${anno}`),
  });
}

interface RecordPaymentData {
  anno: number;
  deadline: string;
  amountPaid: number;
  datePaid: string;
}

export function useRecordPayment() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: (data: RecordPaymentData) =>
      api.post("/taxes/payments", data),
    onSuccess: (_data, variables) => {
      queryClient.invalidateQueries({ queryKey: ["taxes", variables.anno] });
      queryClient.invalidateQueries({ queryKey: ["dashboard"] });
      toast.success("Payment recorded");
      logger.info("Tax payment recorded", variables);
    },
    onError: (error) => {
      toast.error("Failed to record payment");
      logger.error("Tax payment recording failed", error);
    },
  });
}
```
